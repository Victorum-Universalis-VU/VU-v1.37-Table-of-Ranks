	cost = 100
	trigger = {
		has_country_modifier = vu_encourage_abstentionism_aspect_modifier

	effect = {
		add_country_modifier = {
			name = vu_encourage_abstentionism_aspect_modifier

			factor = 2

		}

			factor = 2
			stability = -2
		}
		modifier = {
			factor = 2
			stability = -1
		}
		modifier = {
			factor = 0.5
			stability = 0
		}

			factor = 0.5
			stability = 1
		}
		modifier = {
			factor = 0.5
			stability = 2
		}
	}
}

vu_divorce_consort_aspect = {
	sprite = "GFX_divorce_consort_icon"
	cost = 25
	potential = {
		has_dlc = "Rights of Man"
		government = monarchy
		NOT = { has_reform = iqta }
		NOT = { has_reform = mamluk_government }
	}
	trigger = {
		has_consort = yes
		is_female = no
		has_heir = no
		is_lesser_in_union = no
		has_regency = no
		NOT = { has_country_modifier = VU_anglican_recently_divorced } 
	}
	effect = {
		custom_tooltip = vu_divorce_consort_tooltip
		hidden_effect = {
			add_country_modifier = {
				name = VU_anglican_recently_divorced
				duration = 365
			}
			remove_consort = yes
		}
	}
	ai_will_do = {
		factor = 2
		modifier = {
			factor = 0
			consort_total_stats = {
				amount = 9
			}
		}
	}
}
