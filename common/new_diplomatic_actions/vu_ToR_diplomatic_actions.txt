#########################################################################################
vu_promote_vassal_to_lordship = {
    category = influence
    alert_index = 6
    alert_tooltip = vu_promote_vassal_to_lordship_alert_tooltip

    is_visible = {
        #Checks if the overlord has access to the appropriate subject types
        #has_vu_ToR_subjects = yes
        has_government_attribute = vu_has_ToR_subjects
        ToR_subject_limit_reached = no
        government_rank = 2
        FROM = {
            #FROM has to be a subject of ROOT, a vassal specifically, and also has to be a monarchy
            #(you can't make a republic or a theocracy into a Lordship you dummy)
            is_subject_of = ROOT
            is_subject_of_type = vassal
            government = monarchy
        }
    }

    is_allowed = {
        if = {
            #Limited that the vassal has less than 50 liberty desire
            limit = {
                FROM = {
                    liberty_desire = 50
                }
            }
            custom_trigger_tooltip = {
                tooltip = FROM_IS_REBELLIOUS_TT
                FROM = {
                    liberty_desire = 50
                }
            }
        }
    }

    on_accept = {
        #Made a scripted effect for this, to make this a bit less long and
        #also to make it a bit more efficient as we can re-use these later on
        vu_create_ToR_lordship = yes
        FROM = {
            vu_subject_colouring = yes
        }
    }

    ai_will_do = {
        always = no	#Keeping the AI off this for now, can't be bothered teaching it what to do
    }
}

vu_rein_subject_in = {
    category = influence
    alert_index = 6
    alert_tooltip = vu_rein_subject_in_alert_tooltip

    is_visible = {
        #Checks if the overlord has access to the appropriate subject types
        has_government_attribute = vu_has_ToR_subjects
        if = {
            limit = {
                FROM = {
                    government_rank = 1
                    NOT = {
                        government_rank = 2
                    }
                }
            }
            government_rank = 2
        }
        else_if = {
            limit = {
                FROM = {
                    government_rank = 2
                    NOT = {
                        government_rank = 3
                    }
                }
            }
            government_rank = 3
        }
        else_if = {
            limit = {
                FROM = {
                    government_rank = 3
                    NOT = {
                        government_rank = 4
                    }
                }
            }
            government_rank = 4
        }
        else_if = {
            limit = {
                FROM = {
                    government_rank = 4
                    NOT = {
                        government_rank = 5
                    }
                }
            }
            government_rank = 5
        }
        else_if = {
            limit = {
                FROM = {
                    government_rank = 5
                    NOT = {
                        government_rank = 6
                    }
                }
            }
            government_rank = 6
        }
        else_if = {
            limit = {
                FROM = {
                    government_rank = 6
                    NOT = {
                        government_rank = 7
                    }
                }
            }
            government_rank = 7
        }
        else_if = {
            limit = {
                FROM = {
                    government_rank = 7
                    NOT = {
                        government_rank = 8
                    }
                }
            }
            government_rank = 8
        }
        else_if = {
            limit = {
                FROM = {
                    government_rank = 8
                    NOT = {
                        government_rank = 9
                    }
                }
            }
            government_rank = 9
        }
        FROM = {
            #only works of subjects yours
            is_subject_of = ROOT
            #checks for your subject being any type of a ToR subject
            is_ToR_rein_in_able = yes
        }
    }

    is_allowed = {
        FROM = {
            #cannot be disloyal
            NOT = {
                liberty_desire = 50
            }
            #going descendingly to avoid having to have two checks since government_rank = x is anything above x
            if = {
                limit = {
                    government_rank = 8
                }
                total_development = 100
            }
            else_if = {
                limit = {
                    government_rank = 7
                }
                total_development = 75
            }
            else_if = {
                limit = {
                    government_rank = 6
                }
                total_development = 50
            }
            else_if = {
                limit = {
                    government_rank = 5
                }
                total_development = 40
            }
            else_if = {
                limit = {
                    government_rank = 4
                }
                total_development = 30
            }
            else_if = {
                limit = {
                    government_rank = 3
                }
                total_development = 25
            }
            else_if = {
                limit = {
                    government_rank = 2
                }
                total_development = 15
            }
            else_if = {
                limit = {
                    government_rank = 1
                }
                total_development = 10
            }
        }
    }

    on_accept = {
        if = {
            limit = {
                FROM = {
                    government_rank = 7
                }
            }
            vu_create_ToR_kingdom = yes
        }
        else_if = {
            limit = {
                FROM = {
                    government_rank = 6
                }
            }
            vu_create_ToR_principality = yes
        }
        else_if = {
            limit = {
                FROM = {
                    government_rank = 5
                }
            }
            vu_create_ToR_duchy = yes
        }
        else_if = {
            limit = {
                FROM = {
                    government_rank = 4
                }
            }
            vu_create_ToR_marquisate = yes
        }
        else_if = {
            limit = {
                FROM = {
                    government_rank = 3
                }
            }
            vu_create_ToR_county = yes
        }
        else_if = {
            limit = {
                FROM = {
                    government_rank = 2
                }
            }
            vu_create_ToR_viscounty = yes
        }
        else  = {
            vu_create_ToR_barony = yes
        }
        FROM = {
            vu_subject_colouring = yes
        }
    }

    ai_will_do = {
        always = no	#Keeping the AI off this for now, can't be bothered teaching it what to do
    }
}

vu_release_beylik_subject = { #Added by Prussia to allow breaking of Beylik Vassals for the Ottomans due to populat demand
    category = influence
    require_acceptance = no

    is_visible = {
        FROM = {
            is_subject_of = ROOT
            is_subject_of_type = turkish_tributary_beylik_vassal
        }
    }

    is_allowed = {
        always = yes
    }

    on_accept = {
        FROM = {
            if = {
                limit = { NOT = { liberty_desire = 50 } }
                add_opinion = {
                    who = ROOT
                    modifier = vassal_abandoned
                }
            }
            grant_independence = yes
            add_truce_with = ROOT
            every_owned_province = {
                limit = {
                    OR = {
                        is_claim = ROOT
                        is_permanent_claim = ROOT
                        is_core = ROOT
                    }
                }
                if = {
                    limit = { is_permanent_claim = ROOT }
                    remove_claim = ROOT
                }
                if = {
                    limit = { is_claim = ROOT }
                    remove_claim = ROOT
                }
                if = {
                    limit = { is_core = ROOT }
                    remove_core = ROOT
                }
            }
        }
        add_prestige = -25
    }
    ai_will_do = {
        always = no
    }
}
