has_production_building_trigger = {
	OR = {
		custom_trigger_tooltip = {
			tooltip = has_production_building_trigger_tt
			OR = {
				has_building = workshop
				has_building = counting_house
				has_building = factory				#VU addition
			}
		}
		custom_trigger_tooltip = {
			tooltip = silver_tooltip
			OR = {
				trade_goods = silver
				trade_goods = gold
			}
		}
	}
}
