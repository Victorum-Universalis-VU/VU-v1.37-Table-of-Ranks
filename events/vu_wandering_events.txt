country_event = {
    id = vu_wandering_wenches.1
    title = "vu_wandering_wenches_title"
    picture = "vu_wandering_wenches_eventPicture"
    desc = "vu_wandering_wenches_desc"
    mean_time_to_happen = {
        months = 20
    }
    trigger = {
        OR = {
            any_owned_province = {
                has_province_modifier = vu_wandering_wenches_of_stettin
            }
            AND = {
                NOT = {
                    STE = {
                        owns = 1858
                    }
                }
                owns = 1858
                1858 = {
                    owned_by = ROOT
                    NOT = {
                        has_province_flag = vu_ste_spawned_wenches
                    }
                }
            }
        }
    }
    immediate = {
        if = {
            limit = {
                1858 = {
                    owned_by = ROOT
                    NOT = {
                        has_province_flag = vu_ste_spawned_wenches
                    }
                }
            }
            1858 = {
                set_province_flag = vu_ste_spawned_wenches
            }
        }
    }
    option = {
        #they can go!
        name = vu_wandering_wenches.1a
        custom_tooltip = vu_wandering_wenches.1a.tt
        add_prestige = -10
        add_legitimacy = 10
        add_treasury = 10
        hidden_effect = {
            set_country_flag = denied_the_wenches
            every_owned_province = {
                remove_province_modifier = vu_wandering_wenches_of_stettin
            }
            random_province = {
                limit = {
                    continent = europe
                    religion_group = christian
                    owner = {
                        religion_group = christian
                        NOT = {
                            has_country_flag = denied_the_wenches
                        }
                    }
                    NOT = {
                        OR = {
                            religion = reformed
                            religion = mormon
                            religion = orthodox
                            religion = miaphysite
                            religion = marcionite
                            religion = coptic
                            religion = cathar
                            religion = waldensian
                            religion = nestorian
                            religion = bogomilist_new
                            religion = monothelite
                        }
                    }
                }
                add_province_modifier = {
                    name = vu_wandering_wenches_of_stettin
                    duration = -1
                }
            }
        }
        ai_chance = {
            factor = 1
        }
    }
    option = {
        #they can stay
        name = vu_wandering_wenches.1b
        custom_tooltip = vu_wandering_wenches.1b.tt
        add_prestige = 10
        add_legitimacy = -10
        add_treasury = -10
        if = {
            limit = {
                owns = 1858
            }
            1858 = {
                add_building_construction = {
                    building = cinema
                    speed = 0.25
                    cost = 0
                }
            }
            every_owned_province = {
                remove_province_modifier = vu_wandering_wenches_of_stettin
            }
        }
        else = {
            every_owned_province = {
                limit = {
                    has_province_modifier = vu_wandering_wenches_of_stettin
                }
                add_building_construction = {
                    building = cinema
                    speed = 0.50
                    cost = 0.25
                }
            }

        }

        hidden_effect = {
            set_country_flag = denied_the_wenches
            every_owned_province = {
                remove_province_modifier = vu_wandering_wenches_of_stettin
            }
        }

        ai_chance = {
            factor = 0
        }
    }
}
